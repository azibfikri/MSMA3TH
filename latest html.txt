// Function to calculate Design Rainfall Intensity and Total Depth
function calculateDesignIntensity() {
  const lambda = parseFloat(document.getElementById("idfK").value);
  const kappa = parseFloat(document.getElementById("idfX").value);
  const theta = parseFloat(document.getElementById("idfA").value);
  const eta = parseFloat(document.getElementById("idfN").value);
  const T = parseFloat(document.getElementById("ariYears").value);
  const durationMin = parseFloat(document.getElementById("durationMin").value);
  const d = durationMin / 60; // duration in hours
  
  // Check if all values are valid
  if (isFinite(lambda) && isFinite(kappa) && isFinite(theta) && isFinite(eta) && isFinite(T) && isFinite(d)) {
    // 1. Calculate Intensity (i)
    const i = (lambda * Math.pow(T, kappa)) / Math.pow(d + theta, eta);
    document.getElementById("designIntensity").textContent = i.toFixed(3);

    // 2. Calculate Total Depth (P = i * d)
    const totalDepth = i * d;
    
    // Update the new box below the intensity box
    const depthBox = document.getElementById("totalDepthBox");
    if (depthBox) {
      depthBox.textContent = totalDepth.toFixed(3);
    }
  } else {
    document.getElementById("designIntensity").textContent = "—";
    if (document.getElementById("totalDepthBox")) {
      document.getElementById("totalDepthBox").textContent = "—";
    }
  }
}

// Update autofillConstants to trigger calculation
function autofillConstants() {
  const stationSelect = document.getElementById("stationSelect");
  const selectedOption = stationSelect.options[stationSelect.selectedIndex];
  
  if (selectedOption.value) {
    const lambda = selectedOption.dataset.lambda;
    const kappa = selectedOption.dataset.kappa;
    const theta = selectedOption.dataset.theta;
    const eta = selectedOption.dataset.eta;
    
    document.getElementById("idfK").value = lambda;
    document.getElementById("idfX").value = kappa;
    document.getElementById("idfA").value = theta;
    document.getElementById("idfN").value = eta;
    
    // Automatically calculate after autofill
    calculateDesignIntensity();
  }
}

// Function to update the Depth grid (mm)
function updateDepthPattern() {
  const totalDepthBox = document.getElementById("totalDepthBox");
  const totalDepth = parseFloat(totalDepthBox.textContent);
  const depthGrid = document.getElementById("depthGrid");
  
  // Get all normalized pattern values
  const patternCells = document.querySelectorAll("#patternGrid .patternCell");
  
  // If total depth is not available or pattern is empty, show dashes
  if (!isFinite(totalDepth) || patternCells.length === 0) {
    depthGrid.innerHTML = "";
    
    // Get the time labels from normalized pattern if they exist
    const patternLabels = document.querySelectorAll("#patternGrid div label");
    
    if (patternLabels.length > 0) {
      // Use the same time labels as normalized pattern
      patternLabels.forEach((label) => {
        const wrap = document.createElement("div");
        wrap.innerHTML = `
          <label>${label.textContent}</label>
          <input class="depthCell" type="text" value="---" readonly style="text-align:center; color:var(--muted);" />
        `;
        depthGrid.appendChild(wrap);
      });
    } else {
      // Default placeholder when no pattern exists yet
      for (let i = 0; i < 6; i++) {
        const wrap = document.createElement("div");
        wrap.innerHTML = `
          <label>${i*5}-${(i+1)*5} min</label>
          <input class="depthCell" type="text" value="---" readonly style="text-align:center; color:var(--muted);" />
        `;
        depthGrid.appendChild(wrap);
      }
    }
    
    // Match the grid layout class
    depthGrid.className = document.getElementById("patternGrid").className;
    return;
  }
  
  // Calculate depth = normalized × total depth
  depthGrid.innerHTML = "";
  patternCells.forEach((cell, i) => {
    const normalizedValue = parseFloat(cell.value);
    const depthValue = normalizedValue * totalDepth;
    
    // Get the time label from the corresponding normalized pattern box
    const timeLabel = cell.parentElement.querySelector("label").textContent;
    
    const wrap = document.createElement("div");
    wrap.innerHTML = `
      <label>${timeLabel}</label>
      <input class="depthCell depth-box" type="text" value="${depthValue.toFixed(3)}" readonly style="text-align:center;" />
    `;
    depthGrid.appendChild(wrap);
  });
  
  // Match the grid layout class
  depthGrid.className = document.getElementById("patternGrid").className;
}

  else if (durationMin === 180) durationKey = "30";
  else if (durationMin === 360) durationKey = "60";
  else if (durationMin === 720) durationKey = "120";
  else if (durationMin === 1440) durationKey = "180";

  const pattern = (region && durationKey) ? temporalPatterns[region][durationKey] : null;

  if (pattern) {
    grid.innerHTML = "";
    const binMinutes = durationMin / pattern.length;

    pattern.forEach((value, i) => {
      const wrap = document.createElement("div");
      wrap.innerHTML = `
        <label>${(i * binMinutes).toFixed(0)}-${((i + 1) * binMinutes).toFixed(0)} min</label>
        <input class="patternCell" type="number" step="0.001" value="${value.toFixed(3)}" />
      `;
      grid.appendChild(wrap);
    });

    grid.className = "row";
    if (pattern.length <= 6) grid.className += " six";
    else if (pattern.length <= 12) grid.className += " twelve";
    else grid.className += " twentyfour";
    
    // Update depth pattern after normalized pattern is created
    updateDepthPattern();
  }
}

    document.getElementById("durationMin").value = "";
    updateTemporalPattern(); // This will trigger the dash placeholders
